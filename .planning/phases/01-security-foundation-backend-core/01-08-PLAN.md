1. **Create DailyScoresController** (`backend/app/controllers/api/v1/daily_scores_controller.rb`):
   ```ruby
   module Api
     module V1
       class DailyScoresController < BaseController
         def create
           @daily_score = DailyScore.new(daily_score_params)
           @daily_score.teacher = current_user

           # Verify teacher has access to this student
           student = Student.find(@daily_score.student_id)
           authorize student, :show?  # Teacher must have access to the student
           authorize @daily_score

           if @daily_score.save
             render json: DailyScoreResource.new(@daily_score).serialize, status: :created
           else
             render json: { error: { code: "validation_failed", message: "Validation failed", details: @daily_score.errors } }, status: :unprocessable_entity
           end
         end

         def update
           @daily_score = DailyScore.find(params[:id])
           authorize @daily_score

           if @daily_score.update(daily_score_update_params)
             render json: DailyScoreResource.new(@daily_score).serialize
           else
             render json: { error: { code: "validation_failed", message: "Validation failed", details: @daily_score.errors } }, status: :unprocessable_entity
           end
         end

         private

         def daily_score_params
           params.require(:daily_score).permit(:student_id, :date, :skill_category, :score, :notes)
         end

         def daily_score_update_params
           params.require(:daily_score).permit(:score, :notes)
         end
       end
     end
   end
   ```

2. **Create the materialized view SQL** at `backend/db/views/student_radar_summaries_v01.sql`:
   ```sql
   SELECT
     s.id AS student_id,
     s.name AS student_name,
     s.classroom_id,
     AVG(CASE WHEN ds.skill_category = 0 THEN ds.score END) AS avg_reading,
     AVG(CASE WHEN ds.skill_category = 1 THEN ds.score END) AS avg_math,
     AVG(CASE WHEN ds.skill_category = 2 THEN ds.score END) AS avg_writing,
     AVG(CASE WHEN ds.skill_category = 3 THEN ds.score END) AS avg_logic,
     AVG(CASE WHEN ds.skill_category = 4 THEN ds.score END) AS avg_social,
     COUNT(DISTINCT ds.date) AS total_days_scored,
     MIN(ds.date) AS first_score_date,
     MAX(ds.date) AS last_score_date
   FROM students s
   LEFT JOIN daily_scores ds ON ds.student_id = s.id
   GROUP BY s.id, s.name, s.classroom_id
   ```

     belongs_to :student
     belongs_to :classroom

       Scenic.database.refresh_materialized_view(table_name, concurrently: true, cascade: false)

     def readonly?

     end
   end

5. **Create RefreshRadarSummaryJob:**
   ```ruby
   class RefreshRadarSummaryJob < ApplicationJob
     queue_as :default
     # Deduplicate — only one refresh needed even if many scores submitted
     limits_concurrency to: 1, key: "refresh_radar_summary"

     def perform
       StudentRadarSummary.refresh
     end
   end
   ```
